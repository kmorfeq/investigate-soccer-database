# The following information shows how many players used each feet
# will join player and player attributes tables to in one table to count how many players

ft = read_table("""
SELECT player_feet, count(*) AS feet
  FROM (
    SELECT pl.player_api_id AS player_id, 
        pl.player_name AS player_name, 
        pl.height AS player_height, 
        pl.weight AS player_weight, 
        pla.preferred_foot AS player_feet
    FROM 
        player pl, 
        Player_Attributes pla
    ON 
        pl.player_api_id = pla.player_api_id 
    GROUP BY 
        pl.player_api_id
    ORDER BY 
        pl.player_name)
    GROUP BY 
        player_feet""")
print(ft)
ft.plot.pie( 'feet' , explode = (0 , 0.1), labels= ft.player_feet, startangle=90, shadow = True, autopct='%1.1f%%')
